RMSLE (Root Mean Squared Log Error)의 특징¶

1. 아웃라이이어에 강건해 진다.¶

import numpy as np
from sklearn.metrics import mean_squared_error
In [ ]:
def rmsle(pred, y):
    log_pred=np.log1p(pred)
    log_y=np.log1p(y)
    squared_error=(log_pred-log_y)**2 
    rmsle = np.sqrt(squared_error.mean())
    return rmsle
In [ ]:
y_actual=[60,80,90]
y_pred=[67,78,91]

RMSE = np.sqrt(mean_squared_error(y_pred, y_actual))
print("RMSE :", RMSE)

RMSLE = rmsle(y_pred, y_actual)
print("RMSLE :", RMSLE)
